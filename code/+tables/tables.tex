% Code for each table
\newcommand{\modeltablecode}[3]{%
	\begin{table}[ht] %
	\caption{#1} %
	\centering
	\begin{threeparttable} %
	\input{table#2.tex} %
	#3 % table footnotes
	\end{threeparttable} %
	\end{table} %
	\clearpage
}

% Table footnotes
\newcommand{\modeltablenotes}[2]{%
	\ifthenelse{\equal{#1}{1} \OR \equal{#2}{1}}{%
	\begin{tablenotes} %
	\ifthenelse{\equal{#1}{1}}{%
		\item[$\dagger$] Mean MPC for HtM households, where HtM is defined as assets less than or equal to \$1000.}{}%
    \ifthenelse{\equal{#2}{1}}{%
		\item[*] Where indicated, households are classified as HtM when $a \leq 1\%$ mean annual income.}{}%
    \end{tablenotes}%
    }{}
}
